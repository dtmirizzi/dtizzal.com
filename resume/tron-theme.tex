% === CUSTOM COMMANDS ===

% Section heading with cyan accent line
\newcommand{\sectionheading}[1]{%
  \vspace{6pt}%
  {\color{accentcyan}\fontsize{10}{12}\selectfont\bfseries\texttt{// #1}}%
  \par\vspace{1pt}%
  {\color{sectionrule}\rule{\linewidth}{0.5pt}}%
  \par\vspace{3pt}%
}

% Job title entry - uses etoolbox \ifblank for reliable empty checks
% Args: {title}{company}{dates}{location}{url}
\newcommand{\jobtitle}[5]{%
  {\color{headingtext}\bfseries #1}%
  \ifblank{#2}{}{ {\color{accentcyan}@} \ifblank{#5}{{\color{dimcyan}#2}}{{\color{dimcyan}\href{#5}{\color{dimcyan}#2}}}}%
  \hfill%
  \ifblank{#3}{}{{\color{dimtext}\small #3}}%
  \ifblank{#4}{}{ {\color{dimtext}\small | #4}}%
  \par\vspace{4pt}%
}
